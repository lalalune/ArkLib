import ArkLib.Data.CodingTheory.ProximityGap.LamLeungTwoPow

/-!
# Issue #232 — the multiset form of the 2-power vanishing-sum criterion (O108 seam)

The integer-coefficient / multiset companion to `LamLeungTwoPow.vanishing_sum_antipodal`:
in characteristic zero, an integer combination `∑_{j < 2^(m+1)} c j · ζ^j` of the powers of
a primitive `2^(m+1)`-th root of unity vanishes **iff** the coefficients are antipodally
balanced, `c j = c (j + 2^m)` for all `j < 2^m`. The engine is the same Gauss argument as
the set form: `ζ^(2^m) = −1` folds the sum onto the lower half, and `X^(2^m) + 1` is the
rational minimal polynomial of `ζ`, so the powers `ζ^j` (`j < 2^m`) are linearly
independent over `ℚ` — the folded coefficients must vanish identically.

The corollary `multiset_zeta_pow_sum_eq_zero_iff_antipodal_count` is the literal multiset
form: a multiset of exponents has vanishing root-of-unity sum iff its multiplicity
function is antipodally balanced.
-/

namespace LamLeungTwoPow

open Polynomial Finset

variable {F : Type*} [Field F] [CharZero F]

/-- **The multiset/coefficient form of Lam–Leung at the prime 2** (the O108 seam): an
integer combination of the powers of a primitive `2^(m+1)`-th root of unity vanishes iff
the coefficients are antipodally balanced. -/
theorem zeta_pow_sum_eq_zero_iff_antipodal_balance {m : ℕ} {ζ : F}
    (hζ : IsPrimitiveRoot ζ (2 ^ (m + 1))) (c : ℕ → ℤ) :
    (∑ j ∈ Finset.range (2 ^ (m + 1)), (c j : F) * ζ ^ j = 0) ↔
      ∀ j < 2 ^ m, c j = c (j + 2 ^ m) := by
  classical
  set half := 2 ^ m with hhalf
  have hhn : 2 ^ (m + 1) = half + half := by rw [hhalf]; ring
  have hζh : ζ ^ half = -1 := pow_half_eq_neg_one hζ
  -- fold the sum onto the lower half: ζ^(j + half) = −ζ^j
  have hcollapse : ∑ j ∈ Finset.range (2 ^ (m + 1)), (c j : F) * ζ ^ j
      = ∑ j ∈ Finset.range half, ((c j - c (j + half) : ℤ) : F) * ζ ^ j := by
    rw [hhn, Finset.sum_range_add, ← Finset.sum_add_distrib]
    refine Finset.sum_congr rfl fun j _ => ?_
    have hpow : ζ ^ (half + j) = -ζ ^ j := by
      rw [pow_add, hζh]; ring
    rw [hpow, add_comm half j]
    push_cast
    ring
  constructor
  · -- vanishing forces balance: linear independence of 1, ζ, …, ζ^(half−1) over ℚ
    intro hsum
    set P : ℚ[X] := ∑ j ∈ Finset.range half, C ((c j - c (j + half) : ℤ) : ℚ) * X ^ j
      with hP
    have hPζ : Polynomial.aeval ζ P = 0 := by
      rw [hP, map_sum]
      have hterm : ∀ j ∈ Finset.range half,
          Polynomial.aeval ζ (C ((c j - c (j + half) : ℤ) : ℚ) * X ^ j)
            = ((c j - c (j + half) : ℤ) : F) * ζ ^ j := by
        intro j _
        rw [map_mul, map_pow, Polynomial.aeval_X, Polynomial.aeval_C, map_intCast]
      rw [Finset.sum_congr rfl hterm, ← hcollapse]
      exact hsum
    -- the minimal polynomial X^half + 1 divides P, but deg P < half, so P = 0
    have hdvd : (X ^ half + 1 : ℚ[X]) ∣ P := by
      have hmin := minpoly.dvd ℚ ζ hPζ
      rw [← Polynomial.cyclotomic_eq_minpoly_rat hζ (by positivity)] at hmin
      have hcyc : Polynomial.cyclotomic (2 ^ (m + 1)) ℚ = X ^ half + 1 := by
        rw [Polynomial.cyclotomic_prime_pow_eq_geom_sum Nat.prime_two]
        rw [Finset.sum_range_succ, Finset.sum_range_one, hhalf]
        ring
      rwa [hcyc] at hmin
    have hhpos : 0 < half := by positivity
    have hdegP : P.natDegree < half := by
      have hle : P.natDegree ≤ half - 1 := by
        rw [hP]
        refine Polynomial.natDegree_sum_le_of_forall_le _ _ fun j hj => ?_
        refine (Polynomial.natDegree_C_mul_le _ _).trans ?_
        rw [Polynomial.natDegree_X_pow]
        have := Finset.mem_range.mp hj
        omega
      omega
    have hP0 : P = 0 := by
      by_contra hne
      have hdegle := Polynomial.natDegree_le_of_dvd hdvd hne
      have hdegfac : (X ^ half + 1 : ℚ[X]).natDegree = half := by
        rw [show (X ^ half + 1 : ℚ[X]) = X ^ half + C 1 by rw [map_one]]
        exact Polynomial.natDegree_X_pow_add_C
      rw [hdegfac] at hdegle
      omega
    -- read off the coefficients
    intro j hj
    have hcoeff : P.coeff j = ((c j - c (j + half) : ℤ) : ℚ) := by
      rw [hP, Polynomial.finset_sum_coeff]
      have hterm : ∀ i ∈ Finset.range half,
          (C ((c i - c (i + half) : ℤ) : ℚ) * X ^ i).coeff j
            = if j = i then ((c i - c (i + half) : ℤ) : ℚ) else 0 := by
        intro i _
        rw [Polynomial.coeff_C_mul, Polynomial.coeff_X_pow, mul_ite, mul_one, mul_zero]
      rw [Finset.sum_congr rfl hterm, Finset.sum_ite_eq (Finset.range half) j]
      rw [if_pos (Finset.mem_range.mpr hj)]
    rw [hP0, Polynomial.coeff_zero] at hcoeff
    have hz : (c j - c (j + half) : ℤ) = 0 := by exact_mod_cast hcoeff.symm
    omega
  · -- balance forces vanishing: the folded coefficients are zero
    intro hbal
    rw [hcollapse]
    refine Finset.sum_eq_zero fun j hj => ?_
    have h0 : c j - c (j + half) = 0 :=
      sub_eq_zero.mpr (hbal j (Finset.mem_range.mp hj))
    rw [h0]
    simp

/-- **The literal multiset form**: a multiset `T` of exponents below `2^(m+1)` has
vanishing root-of-unity sum `∑_{j ∈ T} ζ^j = 0` iff its multiplicity function is
antipodally balanced: `count j = count (j + 2^m)` for all `j < 2^m`. -/
theorem multiset_zeta_pow_sum_eq_zero_iff_antipodal_count {m : ℕ} {ζ : F}
    (hζ : IsPrimitiveRoot ζ (2 ^ (m + 1))) {T : Multiset ℕ}
    (hT : ∀ j ∈ T, j < 2 ^ (m + 1)) :
    (T.map fun j => ζ ^ j).sum = 0 ↔
      ∀ j < 2 ^ m, T.count j = T.count (j + 2 ^ m) := by
  classical
  have hsum : (T.map fun j => ζ ^ j).sum
      = ∑ j ∈ Finset.range (2 ^ (m + 1)), ((T.count j : ℤ) : F) * ζ ^ j := by
    rw [Finset.sum_multiset_map_count]
    have hsub : T.toFinset ⊆ Finset.range (2 ^ (m + 1)) := fun j hj =>
      Finset.mem_range.mpr (hT j (Multiset.mem_toFinset.mp hj))
    rw [Finset.sum_subset hsub (fun j _ hj => by
      rw [Multiset.count_eq_zero.mpr (fun h => hj (Multiset.mem_toFinset.mpr h))]
      simp)]
    refine Finset.sum_congr rfl fun j _ => ?_
    rw [nsmul_eq_mul]
    norm_cast
  have hmain := zeta_pow_sum_eq_zero_iff_antipodal_balance hζ fun j => (T.count j : ℤ)
  constructor
  · intro h j hj
    exact_mod_cast hmain.mp (by rw [← hsum]; exact h) j hj
  · intro h
    rw [hsum]
    exact hmain.mpr fun j hj => by exact_mod_cast h j hj

end LamLeungTwoPow
